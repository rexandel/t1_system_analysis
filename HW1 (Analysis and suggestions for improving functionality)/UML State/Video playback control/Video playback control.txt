@startuml
state "Управление воспроизведением видео" as ControlVideo {
  [*] --> Открыта_WEB_страница_с_видео
  
  Открыта_WEB_страница_с_видео --> Воспроизведение_идет: [Нажат "Пробел"] /\n Начать_воспроизведение
  
  state Воспроизведение_идет {
    [*] --> Видео_идет
    Видео_идет --> Воспроизведение_на_паузе: [Нажат_Пробел] /\n Приостановить_воспроизведение
    Видео_идет --> [*]: [Видео_закончилось] /\n Выйти_из_воспроизведения
  }
  
  state Воспроизведение_на_паузе {
    [*] --> Видео_остановлено
    Видео_остановлено --> Воспроизведение_идет: [Нажат_Пробел] /\n Возобновить_воспроизведение
    Видео_остановлено --> [*]: [Видео_закончилось] /\n Выйти_из_воспроизведения
  }
}
@enduml
